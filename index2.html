import java.util.Scanner;

public class ClosestSQQ {
	public static void main(String[] args) {
		Scanner sG = new Scanner(System.in);
		int n = sG.nextInt();
		int a = 1;
		outerLoop: for (int i = 1; i <= n; i++) {
			for (int j = n; j <= n * 100; j++) {
				double k = Math.sqrt(j);
				if ((k * 100 % 10) == 0.00) {
					System.out.println((int) (k * k) + " is the closest perfect square of " + (int) k
							+ " to inputted number " + n + " With the difference of " + (int) Math.abs(n - k * k));
					break outerLoop;
				}
				double k1 = Math.sqrt(n - a);
				a++;
				if ((k1 * 100 % 10) == 0.00) {
					System.out.println((int) (k1 * k1) + " is the closest perfect square of " + (int) k
							+ " to inputted number " + n + " With the difference of " + (int) Math.abs(n - k1 * k1));
					break outerLoop;
				}
			}
		}
	}
}

          import java.util.Scanner;

          public class ReplaceDuplicatesithZero {
          
            public static void main(String... args) {
              Scanner sG = new Scanner(System.in);
              System.out.println("Enter row & clumn size : ");
              int a[][] = new int[sG.nextInt()][sG.nextInt()];
              int c = 0;
          
              for (int i = 0; i < a.length; i++) {
                for (int j = 0; j < a.length; j++) {
                  System.out.print("a[" + i + "][" + j + "] : ");
                  a[i][j] = sG.nextInt();
                }
              }
              for (int i = 0; i < a.length; i++) {
                for (int j = 0; j < a.length; j++) {
                  c = 0;
                  for (int i2 = 0; i2 < a.length; i2++) {
          
                    for (int j2 = 0; j2 < a.length; j2++) {
          
                      if (a[i][j] == a[i2][j2] && (i2 != i || j2 != j)) {
          
                        a[i2][j2] = 0;
                        c++;
                      }
                    }
                  }
                  if (c > 0) {
                    a[i][j] = 0;
                  }
                }
              }
              System.out.println("\nAfter replacing duplicates with zero : ");
              for (int i = 0; i < a.length; i++) {
                for (int j = 0; j < a.length; j++) {
                  System.out.print(a[i][j] + " ");
                }
                System.out.println();
              }
            }
